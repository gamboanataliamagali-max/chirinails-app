    function obtenerTurnos() {
      const data = localStorage.getItem(STORAGE_KEY);
      return data ? JSON.parse(data) : [];
    }

    function guardarTurnos(turnos) {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(turnos));
    }

    // ---------- PANEL ADMIN ----------
    function renderTurnos() {
      const container = document.getElementById("turnosContainer");
      const turnos = obtenerTurnos();

      if (!turnos.length) {
        container.innerHTML = '<div class="no-turnos">TodavÃ­a no hay turnos guardados.</div>';
        return;
      }

      // Ordenar por fecha/hora
      turnos.sort((a, b) => {
        const dA = a.fecha + " " + a.hora;
        const dB = b.fecha + " " + b.hora;
        return dA.localeCompare(dB);
      });

      let html = `
        <table>
          <thead>
            <tr>
              <th>Fecha</th>
              <th>Nombre</th>
              <th>Serv.</th>
              <th>Estado</th>
              <th>Acciones</th>
            </tr>
          </thead>
          <tbody>
      `;

      for (const t of turnos) {
        html += `
          <tr>
            <td>${t.fecha}<br><small>${t.hora}</small></td>
            <td>
              ${t.nombre}<br>
              <small>${t.whatsappCliente}</small>
              ${t.notas ? `<br><small>Notas: ${t.notas}</small>` : ""}
            </td>
            <td>${t.servicio}</td>
            <td>
              <span class="estado-pill ${t.estado === "hecho" ? "hecho" : "pendiente"}">
                ${t.estado === "hecho" ? "Hecho" : "Pendiente"}
              </span>
            </td>
            <td>
              <button class="btn-mini btn-ok" onclick="marcarHecho(${t.id})">âœ”</button>
              <button class="btn-mini btn-del" onclick="borrarTurno(${t.id})">ðŸ—‘</button>
            </td>
          </tr>
        `;
      }

      html += "</tbody></table>";
      container.innerHTML = html;
    }

    window.marcarHecho = function(id) {
      const turnos = obtenerTurnos();
      const idx = turnos.findIndex(t => t.id === id);
      if (idx !== -1) {
        turnos[idx].estado = "hecho";
        guardarTurnos(turnos);
        renderTurnos();
      }
    };

    window.borrarTurno = function(id) {
      if (!confirm("Â¿Seguro que querÃ©s borrar este turno?")) return;
      let turnos = obtenerTurnos();
      turnos = turnos.filter(t => t.id !== id);
      guardarTurnos(turnos);
      renderTurnos();
    };
